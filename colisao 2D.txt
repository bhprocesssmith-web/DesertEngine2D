
bool CheckCollision(SDL_Rect a, SDL_Rect b){

	return(
		a.x < b.x + b.w &&
		a.x + b.w > b.x &&
		a.y < b.y + b.h &&
		a.y + a.h > b.y
	);
}

int main(){
	SDL_Rect player = {100.100.64.64};
	SDL_Rect wall = {300.100.64.64};
	
	if (CheckCollision(player, wall)){
		SDL_Log("colidiu");
	}
	return 0;
}
